// Testing inheritance of roles using the following structure:
//
//                            ┌──────┐                     ┌──────┐
//                ┌───────────┤ root ├──────────┐          │ r-2  │
//                │           └──┬───┘          │          └──┬───┘
//             ┌──┴───┐       ┌──┴───┐       ┌──┴───┐      ┌──┴───┐
//          ┌──┤  A   ├──┐    │  B   │       │  C   │      │  Z   │
//          │  └──────┘  │    └──┬───┘       └──────┘      └──────┘
//          │            │       │
//       ┌──┴───┐    ┌───┴──┐ ┌──┴───┐                                   
//    ┌──┤  AA  ├┐  ┌┤  AB  │ │  BA  │
//    │  └──────┘|  |└──────┘ └──────┘
// ┌──┴───┐    ┌─┴──┴─┐
// │ AAA  │  ┌─┤  X   ├─┐
// └──────┘  │ └──────┘ │
//       ┌───┴──┐    ┌──┴───┐
//       │  XA  │    │  XB  │
//       └──────┘    └──────┘

:contract HierarchyOrgChart(std)

:role root
:role A(root)
:role B(root)
:role C(root)
:role AA(A)
:role AB(A)
:role BA(B)
:role X(AA,AB)
:role XA(X)
:role XB(X)
:role AAA(AA)
:role root2
:role Z(root2)
